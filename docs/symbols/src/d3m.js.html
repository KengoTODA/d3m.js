  9   function D3M(context, width, height) {
 10     this.context = context;
 11     this.width = width != null ? width : 320;
 12     this.height = height != null ? height : 320;

 14   /**
 15   入力された x, y, z の距離 (ベクトル) の絶対値 ( sqrt(x*x + y*y + z*z) ) を返す関数です。
 16   
 17   z 座標を省略すれば平面座標での距離を求めることができます。
 18   @param {Number} x
 19   @param {Number} [y]
 20   @param {Number} [z]
 21   */
 22   D3M.prototype.d3dist = function(x, y, z) {
 23     if (y == null) {
 24       y = 0;
 25     }
 26     if (z == null) {
 27       z = 0;
 28     }
 29     return Math.sqrt(x * x + y * y + z * z);
 30   };
 31   /**
 32   平面上のベクトル (あるいは位置) x0, y0 を、原点を中心に va [ラジアン] 回転させたベクトルを求めます。計算結果は、x, y をプロパティに持つオブジェクトとして返されます。
 33   
 34   角度は、ラジアン単位 (2π ≒ 6.2831853 を一回転とする角度単位) で入力します。
 35   @param {Number} x0 回転させるベクトルのx方向の大きさ
 36   @param {Number} y0 回転させるベクトルのy方向の大きさ
 37   @param {Number} va 回転量（ラジアン）
 38   */
 39   D3M.prototype.d3rotate = function(x0, y0, va) {
 40     return {
 41       x: x0 * Math.cos(va) - y0 * Math.sin(va),
 42       y: x0 * Math.sin(va) + y0 * Math.cos(va)
 43     };
 44   };
 45   D3M.prototype.d3vrotate = function(x0, y0, z0, vx, vy, vz, va) {
 46     var ax, ay, az, cos, l_cos, r, sin;
 47     r = this.d3dist(vx, vy, vz);
 48     ax = vx / r;
 49     ay = vy / r;
 50     az = vz / r;
 51     sin = Math.sin(va);
 52     cos = Math.cos(va);
 53     l_cos = 1.0 - cos;
 54     return {
 55       x: (ax * ax * l_cos + cos) * x0 + (ax * ay * l_cos - az * sin) * y0 + (az * ax * l_cos - ax * sin) * z0,
 56       y: (ax * ay * l_cos + az * sin) * x0 + (ay * ay * l_cos + cos) * y0 + (ay * az * l_cos - ax * sin) * z0,
 57       z: (az * ax * l_cos - ay * sin) * x0 + (ay * az * l_cos + ax * sin) * y0 + (az * az * l_cos + cos) * z0
 58     };
 59   };
 60   D3M.prototype.d3setlocal = function(LGmpx, LGmpy, LGmpz, LGm00, LGm10, LGm20, LGm01, LGm11, LGm21, LGm02, LGm12, LGm22) {
 61     if (LGmpx == null) {
 62       LGmpx = 0;
 63     }
 64     if (LGmpy == null) {
 65       LGmpy = 0;
 66     }
 67     if (LGmpz == null) {
 68       LGmpz = 0;
 69     }
 70     if (LGm00 == null) {
 71       LGm00 = 1;
 72     }
 73     if (LGm10 == null) {
 74       LGm10 = 0;
 75     }
 76     if (LGm20 == null) {
 77       LGm20 = 0;
 78     }
 79     if (LGm01 == null) {
 80       LGm01 = 0;
 81     }
 82     if (LGm11 == null) {
 83       LGm11 = 1;
 84     }
 85     if (LGm21 == null) {
 86       LGm21 = 0;
 87     }
 88     if (LGm02 == null) {
 89       LGm02 = 0;
 90     }
 91     if (LGm12 == null) {
 92       LGm12 = 0;
 93     }
 94     if (LGm22 == null) {
 95       LGm22 = 1;
 96     }
 97     this.LGSm00 = this.GSm00 * LGm00 + this.GSm10 * LGm01;
 98     this.LGSm10 = this.GSm00 * LGm10 + this.GSm10 * LGm11;
 99     this.LGSm20 = this.GSm00 * LGm20 + this.GSm10 * LGm21;
100     this.LGSmpx = this.GSm00 * LGmpx + this.GSm10 * LGmpy + this.GSmpx;
101     this.LGSm01 = this.GSm01 * LGm00 + this.GSm11 * LGm01 + this.GSm21 * LGm02;
102     this.LGSm11 = this.GSm01 * LGm10 + this.GSm11 * LGm11 + this.GSm21 * LGm12;
103     this.LGSm21 = this.GSm01 * LGm20 + this.GSm11 * LGm21 + this.GSm21 * LGm22;
104     this.LGSmpy = this.GSm01 * LGmpx + this.GSm11 * LGmpy + this.GSm21 * LGmpz + this.GSmpy;
105     this.LGSm02 = this.GSm02 * LGm00 + this.GSm12 * LGm01 + this.GSm22 * LGm02;
106     this.LGSm12 = this.GSm02 * LGm10 + this.GSm12 * LGm11 + this.GSm22 * LGm12;
107     this.LGSm22 = this.GSm02 * LGm20 + this.GSm12 * LGm21 + this.GSm22 * LGm22;
108     this.LGSmpz = this.GSm02 * LGmpx + this.GSm12 * LGmpy + this.GSm22 * LGmpz + this.GSmpz;
109   };
110   D3M.prototype.d3setcam = function(cpx, cpy, cpz, ppx, ppy, ppz, ppv) {
111     var ax, ay, az, cos0, cos1, r0, r1, sin0, sin1;
112     if (cpx == null) {
113       cpx = 0;
114     }
115     if (cpy == null) {
116       cpy = 0;
117     }
118     if (cpz == null) {
119       cpz = 0;
120     }
121     if (ppx == null) {
122       ppx = 0;
123     }
124     if (ppy == null) {
125       ppy = 0;
126     }
127     if (ppz == null) {
128       ppz = 0;
129     }
130     if (ppv == null) {
131       ppv = 1;
132     }
133     this.wincx = this.width / 2;
134     this.wincy = this.height / 2;
135     ax = cpx - ppx;
136     ay = cpy - ppy;
137     az = cpz - ppz;
138     r0 = Math.sqrt(ax * ax + ay * ay);
139     r1 = Math.sqrt(r0 * r0 + az * az);
140     if (r0 !== 0.0) {
141       cos0 = -ax / r0;
142       sin0 = -ay / r0;
143     }
144     if (r1 !== 0.0) {
145       cos1 = r0 / r1;
146       sin1 = az / r1;
147     }
148     az = ppv / (0.01 + this.height);
149     this.GSm00 = sin0;
150     this.GSm10 = -cos0;
151     this.GSm01 = cos0 * cos1 * az;
152     this.GSm11 = sin0 * cos1 * az;
153     this.GSm21 = -sin1 * az;
154     this.GSm02 = cos0 * sin1;
155     this.GSm12 = sin0 * sin1;
156     this.GSm22 = cos1;
157     this.GSmpx = -(this.GSm00 * cpx + this.GSm10 * cpy);
158     this.GSmpy = -(this.GSm01 * cpx + this.GSm11 * cpy + this.GSm21 * cpz);
159     this.GSmpz = -(this.GSm02 * cpx + this.GSm12 * cpy + this.GSm22 * cpz);
160     this.d3setlocal(0, 0, 0, 1, 0, 0, 0, 1, 0, 0, 0, 1);
161   };
162   D3M.prototype.d3trans = function(x, y, z) {
163     this.dz = this.LGSm01 * x + this.LGSm11 * y + this.LGSm21 * z + this.LGSmpy;
164     this.df = false;
165     if (this.dz > 0) {
166       this.dx = this.wincx + (this.LGSm00 * x + this.LGSm10 * y + this.LGSm20 * z + this.LGSmpx) / this.dz;
167       this.dy = this.wincy - (this.LGSm02 * x + this.LGSm12 * y + this.LGSm22 * z + this.LGSmpz) / this.dz;
168       this.df = this.dx < 8000 && this.dy < 8000;
169     }
170   };
171   D3M.prototype.d3vpos = function(x, y, z) {
172     var _ref;
173     _ref = [this.dx, this.dy, this.df], this.ex = _ref[0], this.ey = _ref[1], this.ef = _ref[2];
174     this.d3trans(x, y, z);
175   };
176   D3M.prototype.d3getpos = function(x, y, z) {
177     this.d3vpos(x, y, z);
178     if (this.df) {
179       return {
180         x: this.dx,
181         y: this.dy
182       };
183     }
184     return null;
185   };
186   D3M.prototype.d3pos = function(x, y, z) {
187     this.d3vpos(x, y, z);
188     if (this.df) {
189       this.context.moveTo(this.dx, this.dy);
190     }

222   D3M.prototype.d3line = function(ppx, ppy, ppz, ssx, ssy, ssz) {
223     var ax, ay, az, bY, bx, bz, cnt, cx, cy, cz, _ref, _ref2, _ref3, _ref4, _ref5, _ref6;
224     if (ppx == null) {
225       ppx = 0;
226     }
227     if (ppy == null) {
228       ppy = 0;
229     }
230     if (ppz == null) {
231       ppz = 0;
232     }
233     if (ssx == null) {
234       ssx = 0;
235     }
236     if (ssy == null) {
237       ssy = 0;
238     }
239     if (ssz == null) {
240       ssz = 0;
241     }
242     this.d3vpos(ssx, ssy, ssz);
243     this.d3vpos(ppx, ppy, ppz);
244     if (this.df && this.ef) {
245       this.context.moveTo(this.ex, this.ey);
246       this.context.lineTo(this.dx, this.dy);
247       return;
248     }
249     if (this.df || this.ef) {
250       if (this.df) {
251         this.context.moveTo(this.dx, this.dy);
252         _ref = [ppx, ppy, ppz], ax = _ref[0], ay = _ref[1], az = _ref[2];
253         _ref2 = [ssx, ssy, ssz], bx = _ref2[0], bY = _ref2[1], bz = _ref2[2];
254       } else {
255         this.context.moveTo(this.ex, this.ey);
256         _ref3 = [ssx, ssy, ssz], ax = _ref3[0], ay = _ref3[1], az = _ref3[2];
257         _ref4 = [ppx, ppy, ppz], bx = _ref4[0], bY = _ref4[1], bz = _ref4[2];
258       }
259       for (cnt = 0; cnt <= 9; cnt++) {
260         cx = (ax + bx) / 2;
261         cy = (ay + bY) / 2;
262         cz = (az + bz) / 2;
263         this.d3trans(cx, cy, cz);
264         if (this.df) {
265           _ref5 = [cx, cy, cz], ax = _ref5[0], ay = _ref5[1], az = _ref5[2];
266           this.context.lineTo(this.dx, this.dy);
267         } else {
268           _ref6 = [cx, cy, cz], bx = _ref6[0], bY = _ref6[1], bz = _ref6[2];
269         }
270       }
271     }
272   };
273   D3M.prototype.d3arrow = function(x1, y1, z1, x2, y2, z2) {
274     var a, bY, bx, r;
275     this.d3line(x1, y1, z1, x2, y2, z2);
276     if (this.df && this.ef) {
277       a = Math.atan2(this.dy - this.ey, this.dx - this.ex);
278       this.d3vpos((x1 * 6 + x2) / 7, (y1 * 6 + y2) / 7, (z1 * 6 + z2) / 7);
279       r = this.d3dist(x1 - x2, y1 - y2, z1 - z2) / this.dz / 25;
280       bx = Math.cos(a) * r;
281       bY = Math.sin(a) * r;
282       this.context.moveTo(this.dx - bY, this.dy + bx);
283       this.context.lineTo(this.ex, this.ey);
284       this.context.lineTo(this.dx + bY, this.dy - bx);
285     }
286   };

301   D3M.prototype.d3circle = function(x, y, z, r, flg) {
